{-# LANGUAGE DerivingVia #-}

import Control.Monad.Writer

data Message = Message

class Monad m => MonadSendMessage m where
  sendMessage :: Message -> m ()

instance MonadSendMessage IO where
  sendMessage _ = {- use your imagination :) -} pure ()

newtype TestMsg a = TestMsg (Writer [Message] a)
  deriving (Functor, Applicative, Monad, MonadWriter [Message]) via Writer [Message]

data Context = Context {}

newtype Runner a = Runner (ReaderT Context IO a)
